def load_data(opts = {})
	@opts = GenericBusGijonModel::DEFAULT_OPTS.merge(opts)

	case @opts.fetch(:load_data, nil)
		when :static
			load_static_data(@opts)
		when :dynamic
			load_dynamic_data(@opts)
		when :all
			load_static_data(@opts)
			load_dynamic_data(@opts)
		else
			raise RequiredParamMissingException.new(:load_data)
	end
end
